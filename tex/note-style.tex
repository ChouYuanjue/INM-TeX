\newcommand{\notebadge}[1]{%
  \tikz[baseline=(badge.base)]\node[
    rounded corners=3pt,
    draw=purple!70!black,
    fill=purple!6,
    inner xsep=7pt,
    inner ysep=3pt,
    line width=.45pt
  ] (badge) {\sffamily\bfseries\footnotesize\color{purple!70!black} #1};%
}

\NewDocumentCommand{\notechapter}{m m}{%
  \cleardoublepage
  \refstepcounter{chapter}%
  \setcounter{section}{0}%
  \setcounter{subsection}{0}%
  \setcounter{theorem}{0}%
  \phantomsection
  \addcontentsline{toc}{chapter}{#1}%
  \chaptermark{#2}%
  \markboth{#2}{#2}%
  \thispagestyle{plain}%
  \vspace*{0.6em}%
  \noindent
  \begin{minipage}[t]{0.66\textwidth}
    {\usekomafont{chapter}\raggedright #2\par}%
  \end{minipage}%
  \hfill
  \raisebox{0.35em}{\notebadge{#1}}%
  \par\vspace{0.8em}
  \noindent{\color{purple!60!black}\rule{\textwidth}{0.7pt}}%
  \par\vspace{1.0em}%
}
